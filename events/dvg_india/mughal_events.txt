namespace = mughal

# Startup event
mughal.1 = {
	type = country_event
	placement = ROOT
	
	title = mughal.1.t
	desc = mughal.1.d
	flavor = mughal.1.f
	
	event_image = {
		video = "asia_politician_parliament_motion"
	}

	on_created_soundeffect = "event:/SFX/UI/Alerts/event_appear"

	icon = "gfx/interface/icons/event_icons/event_portrait.dds"
	
	duration = 3

	trigger = {
		c:MUG ?=  this
		NOT = { has_variable = mughal_salaf_var}
		any_scope_character = {
			has_variable = is_salaf
		}
	}

	immediate = {
		set_variable = mughal_salaf_var
	}

	option = {
		name = salaf.1.a
		default_option = yes
		custom_tooltip = {
			text = salaf_tt
		}
	}
}
mughal.2 = {
	type = country_event
	placement = ROOT
	
	title = mughal.2.t
	desc = mughal.2.d
	flavor = mughal.2.f
	
	event_image = {
		video = "middleeast_upperclass_party"
	}

	on_created_soundeffect = "event:/SFX/UI/Alerts/event_appear"

	icon = "gfx/interface/icons/event_icons/event_portrait.dds"
	
	duration = 3

	trigger = {
		c:MUG ?=  this
		c:MUG = {
		has_variable = mughal_salaf_var
		NOT = {
		any_scope_character = {
				has_variable = is_salaf
			}
		}
		NOT = {has_variable = mughal_civil_war_happened}
		}
	}
	
	immediate = {
		random_scope_character = {
			limit = {
				has_variable = is_qutb
			}
			save_scope_as = qutb_scope
		}
		random_scope_character = {
			limit = {
				has_variable = is_nasir
			}
			save_scope_as = nasir_scope
		}
	}

	option = {
		name = mughal.2.a
		default_option = yes
		scope:qutb_scope = {
			set_character_as_ruler = yes
		}
		scope:nasir_scope = {
			set_variable = mah_throne_var
		}
	}
	option = {
		name = mughal.2.b
		scope:nasir_scope = {
			set_character_as_ruler = yes
		}
		scope:qutb_scope = {
			set_variable = mah_throne_var
		}
	}
}
