# Annex an Italian country
italian_unification.2 = {
	type = country_event
	placement = ROOT

	title = italian_unification.2.t
	desc = italian_unification.2.d
	flavor = italian_unification.2.f

	event_image = {
		video = "unspecific_signed_contract"
	}

	on_created_soundeffect = "event:/SFX/UI/Alerts/event_appear"

	icon = "gfx/interface/icons/event_icons/waving_flag.dds"

	duration = 1

	trigger = {
		OR = {
			NOT = { has_journal_entry = je_italian_confederation }
			has_global_variable = ITC_council_succes
			has_global_variable = ITC_council_failed
		}
		any_neighbouring_state = {
			owner = {
				trigger_if = {
					limit = {
						has_technology_researched = pan-nationalism
					}
					OR = {
						country_has_primary_culture = cu:north_italian
						country_has_primary_culture = cu:south_italian
						country_has_primary_culture = cu:central_italian #DVG
						country_has_primary_culture = cu:sardinian #DVG
						country_has_primary_culture = cu:venetian #DVG
					}
				}
				trigger_else_if = {
					limit = {
						ROOT = {
							OR = {
								country_has_primary_culture = cu:north_italian
								country_has_primary_culture = cu:central_italian
							}
						}
					}
					OR = {
						country_has_primary_culture = cu:north_italian
						country_has_primary_culture = cu:central_italian
					}
				}
				trigger_else_if = {
					limit = {
						ROOT = {
							country_has_primary_culture = cu:south_italian
						}
					}
					country_has_primary_culture = cu:south_italian
				}
				OR = {
					is_subject = no
					is_subject_of = ROOT
				}
				OR = {
					ROOT = { has_journal_entry = je_italian_confederation }
					has_diplomatic_pact = {
						who = ROOT
						type = customs_union
					}
				}
				country_rank <= rank_value:minor_power
				relations:root >= relations_threshold:amicable
				is_player = no
				NOT = { has_variable = italian_minor_dont_annex }
			}
		}
	}

	immediate = {
		random_neighbouring_state = {
			limit = {
				owner = {
					trigger_if = {
						limit = {
							has_technology_researched = pan-nationalism
						}
						OR = {
							country_has_primary_culture = cu:north_italian
							country_has_primary_culture = cu:south_italian
							country_has_primary_culture = cu:central_italian #DVG
							country_has_primary_culture = cu:sardinian #DVG
							country_has_primary_culture = cu:venetian #DVG
						}
					}
					trigger_else_if = {
						limit = {
							ROOT = {
								OR = {
									country_has_primary_culture = cu:north_italian
									country_has_primary_culture = cu:central_italian
								}
							}
						}
						OR = {
							country_has_primary_culture = cu:north_italian
							country_has_primary_culture = cu:central_italian
						}
					}
					trigger_else_if = {
						limit = {
							ROOT = {
								country_has_primary_culture = cu:south_italian
							}
						}
						country_has_primary_culture = cu:south_italian
					}
					OR = {
						is_subject = no
						is_subject_of = ROOT
					}
					OR = {
						ROOT = { has_journal_entry = je_italian_confederation }
						has_diplomatic_pact = {
							who = ROOT
							type = customs_union
						}
					}
					country_rank <= rank_value:minor_power
					relations:root >= relations_threshold:amicable
					is_player = no
					NOT = { has_variable = italian_minor_dont_annex }
				}
			}
			owner = {
				save_scope_as = italian_annex_country
			}
		}
	}

	option = {
		name = italian_unification.2.a
		default_option = yes
		create_incident = {
			country = scope:italian_annex_country
			value = 10
		}
		annex = scope:italian_annex_country
		ai_chance = {
			base = 1
		}
	}
	option = {
		name = italian_unification.2.b
		change_relations = {
			country = scope:italian_annex_country
			value = 20
		}
		scope:italian_annex_country = {
			set_variable = {
				name = italian_minor_dont_annex
				months = normal_modifier_time
			}
		}
		ai_chance = {
			base = 0
		}
	}
}
