#Qing Have Conquered Beijing
dvg_qing.2 = {
	type = country_event
	placement = ROOT

	title = dvg_qing.2.t
	desc = dvg_qing.2.d
	flavor = dvg_qing.2.f

	event_image = {
		video = "unspecific_military_parade"
	}

	on_created_soundeffect = "event:/SFX/UI/Alerts/event_appear"
	on_opened_soundeffect = "event:/SFX/Events/unspecific/military_parade"

	icon = "gfx/interface/icons/event_icons/event_portrait.dds"

	duration = 3

	trigger = {

	}
	immediate = {
		c:SHU = { save_scope_as = scope_SHU}
		c:DZU = { save_scope_as = scope_DZU}
		c:KOR = { save_scope_as = scope_KOR}
	}

	option = {
		name = dvg_qing.2.a

		add_journal_entry = {
			type = je_qing_conquest
		}
		s:STATE_BEIJING = {

			add_homeland = cu:manchu		
		}
		add_primary_culture = cu:han
		set_capital = STATE_BEIJING
		custom_tooltip = {
		text = claim_china_tt
		every_state_region = {
			limit = {
			OR = {
				is_homeland = cu:han
				}
			}
			add_claim = ROOT
		}
		}
		set_variable = China_unifer
		if = {
			limit = {
				is_ai = yes
			}
			set_strategy = ai_strategy_unify_china
		}
	}
}

dvg_qing.3 = {
	type = country_event
	placement = ROOT

	title = dvg_qing.3.t
	desc = dvg_qing.3.d
	flavor = dvg_qing.3.f

	event_image = {
		video = "gfx/event_pictures/southamerica_war_civilians.bk2"
	}

	on_created_soundeffect = "event:/SFX/UI/Alerts/event_appear"
	on_opened_soundeffect = "event:/SFX/Events/unspecific/military_parade"

	icon = "gfx/interface/icons/event_icons/event_portrait.dds"

	duration = 3

	trigger = {

	}

	option = {
		name = dvg_qing.3.a

		s:STATE_SHENGJING = {
			remove_claim = c:SHU
		}
		s:STATE_BEIJING = {
			remove_claim = c:CHI
		}

		if = {
			limit = { owns_entire_state_region = STATE_SHENGJING }

			s:STATE_SHENGJING = {

				remove_homeland = cu:han				
			}
		}
		activate_law = law_type:law_national_supremacy

		ig:ig_landowners = { 
			set_interest_group_name = ig_scholar_officials
			remove_ideology = ideology_scholar_paternalistic
			remove_ideology = ideology_confucian
			add_ideology = ideology_paternalistic
		}


		add_loyalists = {
			value = .3
			culture = cu:manchu
		}
	}
}
