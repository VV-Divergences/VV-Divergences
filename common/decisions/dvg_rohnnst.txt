dvg_integrate_new_lands_rohnnst = {

	is_shown = {
		exists = c:RHN
		root = c:RHN
		root = {
			OR = {
				AND = {
					has_state_in_state_region = STATE_WENYUAN
					s:STATE_WENYUAN = {
						NOT = {
							is_homeland = cu:rohnnst
						}
					}					
				}
				AND = {
					has_state_in_state_region = STATE_BAIZEHU
					s:STATE_BAIZEHU = {
						NOT = {
							is_homeland = cu:rohnnst
						}
					}					
				}
				AND = {
					has_state_in_state_region = STATE_AIOUEZ
					s:STATE_AIOUEZ = {
						NOT = {
							is_homeland = cu:rohnnst
						}
					}					
				}
			}
		}
		NOT = {
			has_law = law_type:law_multicultural
		}
	}
	
	possible = {
		root = {
			OR = {
				AND = {
					owns_entire_state_region = STATE_WENYUAN
					s:STATE_WENYUAN = {
						NOT = {
							is_homeland = cu:rohnnst
						}
					}
				}
				AND = {
					owns_entire_state_region = STATE_BAIZEHU
					s:STATE_BAIZEHU = {
						NOT = {
							is_homeland = cu:rohnnst
						}
					}
				}
				AND = {
					owns_entire_state_region = STATE_AIOUEZ
					s:STATE_AIOUEZ= {
						NOT = {
							is_homeland = cu:rohnnst
						}
					}
				}
			}
		}	
	}

    when_taken = {
		IF = {
			limit = {
				c:RHN = {
					owns_entire_state_region = STATE_BAIZEHU
					}
				s:STATE_BAIZEHU = {
					NOT = {
						is_homeland = cu:rohnnst
					}
				}
			}
			s:STATE_BAIZEHU = {
				add_homeland = cu:rohnnst
			}
		}
		IF = {
			limit = {
				c:RHN = {
					owns_entire_state_region = STATE_WENYUAN
					}
				s:STATE_WENYUAN = {
					NOT = {
						is_homeland = cu:rohnnst
					}
				}
			}
			s:STATE_WENYUAN = {
				add_homeland = cu:rohnnst
			}
		}
		IF = {
			limit = {
				c:RHN = {
					owns_entire_state_region = STATE_AIOUEZ
					}
				s:STATE_AIOUEZ = {
					NOT = {
						is_homeland = cu:rohnnst
					}
				}
			}
			s:STATE_AIOUEZ = {
				add_homeland = cu:rohnnst
			}
		}
    }
}

	is_shown = {
		exists = c:RHN
		root = c:RHN
		NOT = {
			is_subject = yes
		}	
	}
	
	possible = {
		c:RHN = {
			is_country_type = recognized
			owns_entire_state_region = STATE_AIOUEZ

				NOT = {
