
dvg_bharat_subcontinent = {
	is_shown = {

	
		any_primary_culture = {
			has_discrimination_trait = south_asian_heritage
		}

		any_primary_culture = {
			has_discrimination_trait = hindustani
		}
		NOT = {
			has_global_variable = dvg_bharata
		}
	}

    possible = {
		OR = {
			country_tier = empire
			country_tier = hegemony
		}
	}

    when_taken = {

		create_incident = {
			country = ROOT
			value = 20
		}
		
		every_state_region = {
			limit = {
				region = sr:region_north_india
			}
			add_claim = ROOT
		}
		
		set_global_variable = dvg_bharata

}
ai_chance = {
		base = 100
	}
}

dvg_sikh_embrace = {
	is_shown = {
	
		has_variable = dvg_sikh_accepted
		NOT = {
			cu:panjabi = {
				culture_is_discriminated_in = ROOT 
		   }
		   has_variable = dvg_convert_sikh
		   exists = c:PAN
		}
	}

    possible = {
		
		NOT = {
			has_modifier = decision_cooldown
		}
		any_interest_group = {
			leader = {
				has_religion = rel:sikh
			}
			is_powerful = yes
		}
		has_law = law_type:law_religious_autonomys
	}

    when_taken = {

		trigger_event = {
			id = dvg_afghanistan.4
		}

	}

		base = 100
