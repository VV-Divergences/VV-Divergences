dvg_sandakan = {
	is_shown = {
		exists = c:JAP
		exists = c:SDK
		OR = {
			c:SDK = ROOT
			c:JAP = ROOT
		}
		is_player = yes 
		c:SDK = {
			NOR = {
				has_modifier = resource_extraction_charter_modifier
				has_modifier = religious_missions_colonial_modifier
				has_modifier = settlement_campaigns_modifier
			}

		}
	}

    possible = {
		has_technology_researched = civilizing_mission
	}

    when_taken = {
		trigger_event = {
			id = dvg_japan.8
		}
    }

	ai_chance = {
		base = 100
	}
}

dvg_embrace_ainu = {
	is_shown = {
		any_scope_state = {

			state_region = {

				is_homeland = ainu
			}
			literacy_rate >= 0.70 
			
		}
	}

    possible = {
		any_interest_group = {
			is_powerful = yes
			law_stance = {
				law = law_type:law_cultural_exclusion
				value > neutral
			}
		}
	}

    when_taken = {
		add_primary_culture = cu:ainu
    }

	ai_chance = {
		base = 100
	}
}
dvg_ditch_shinto = {
	is_shown = {
		country_has_state_religion = rel:shinto
		has_law = law_type:law_total_separation

	}

    possible = {
		
		NOT = {
			has_law = law_type:law_theocracy
			has_law = law_type:law_monarchy
		}
	}

    when_taken = {
		set_state_religion = rel:mahayana

		every_country = {
			limit = {
				country_has_state_religion = rel:mahayana
			}
			change_relations = {
				country = ROOT
				value = 10
			}
		}
    }

	ai_chance = {
		base = 100
	}
}

dvg_annex_siberia_outsider = {
	is_shown = {
	
		NOT = { is_country_type = decentralized }
		exists = c:SIB

		any_state = {
			owner = c:SIB
			has_claim_by = ROOT
		}

		has_interest_marker_in_region = sr:region_west_siberia
		
		
	}

    possible = {


		NOT = {
			has_truce_with = c:SIB
		}
	
	}

    when_taken = {

		random_state = {
			limit = {
				owner = c:SIB
				any_neighbouring_state = {
					owner = ROOT
				}
				NOT = {
					owner = ROOT
				}
			}

			set_state_owner = ROOT
		}
		
		
		create_truce = {
			country = SIB
			months = 60
		}
		create_incident = {
			country = ROOT
			value = 3
		}
    }

	ai_chance = {
		base = 100
	}
}
