	group = je_group_internal_affairs
	
	complete = {
		OR = {
			cu:quechua = {
				NOT = { culture_is_discriminated_in = ROOT }
			}
			NOT = {
				exists = c:TWT
			}
		}
	}
	
	on_yearly_pulse = {
		events = {
			dvg_caladero.4
			dvg_caladero.5
		}
	}
	
	fail = {
		OR = {
			cu:quechua = {

		trigger_event = { id = dvg_caladero.6 }

	}
	
	weight = 100
	should_be_pinned_by_default = yes
}
