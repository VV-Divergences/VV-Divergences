je_llapa_succession = {
	icon = "gfx/interface/icons/invention_icons/joint_stock_companies.dds"
	
	group = je_group_internal_affairs
	
	complete = {
		ig:ig_intelligentsia = {
						is_powerful = yes
					}
		
	}

    on_complete = {
		trigger_event = { id = dvg_tawantisuyu_reconquest.6 }
	}
	fail = {
		AND = {
		ig:ig_armed_forces = {
						is_powerful = yes
					}
		ig:ig_intelligentsia = {
						is_powerful = no
					}
		
		}
		
	}
	on_fail = {
		trigger_event = { id = dvg_tawantisuyu_reconquest.7 }
	}
	invalid = {
	NOT = {
		any_character = {
					has_variable = is_llapa_inka
				}
		}
	}
	
	weight = 100
	should_be_pinned_by_default = yes
}

je_beornia_revolt = {
	icon = "gfx/interface/icons/event_icons/event_military.dds"
	
	group = je_group_internal_affairs

	possible = {
		
	}

	immediate = {
		set_variable = {
			name = beornia_hope_var
			value = 0
		}
	}

	on_monthly_pulse = {
		effect = {
			if = {
				limit = {
					cu:beornian = {
			 culture_is_discriminated_in = ROOT 
			}	
				}
				change_variable = {
					name = armenia_var
					add = 1
				}
			}
			if = {
				limit = {
					cu:beornian = {
				culture_secession_progress = {
					target = ROOT
					value > 0.9
				}	
				}
				
						}
				change_variable = {
					name = armenia_var
					add = 24
				}
			}
		}
	}

	complete = {
		trigger_if = {
			limit = { exists = var:beornia_hope_var }
			scope:journal_entry = { is_goal_complete = yes }
		}
	}
	goal_add_value = {
		value = 100
	}

    on_complete = {
		trigger_event = { id = dvg_hbe.5 }
	}

	progressbar = yes
	weight = 100
	should_be_pinned_by_default = yes
}
