je_nov_main = {
	icon = "gfx/interface/icons/event_icons/event_skull.dds"
	
	group = je_group_internal_affairs


	immediate = {
		set_variable = {
			name = nov_var
			value = 0
		}
		set_variable = {
			name = nov_serfdom
			value = 0
		}
	}
	
	complete = {
		scope:journal_entry = {
			is_goal_complete = yes
		}
	}

	on_complete = {
		trigger_event = { id = dvg_nov.2 }
	}

	fail = {
		OR = {
			NOT = { has_law = law_type:law_presidential_republic }
		}
	}

	on_fail = {
		trigger_event = { id = dvg_nov.4 }
	}
	on_timeout = {
		trigger_event = { id = dvg_nov.4 }
	}

	modifiers_while_active = {
		neutrality
	}

	on_monthly_pulse = {
		effect = {
			if = {
				limit = {
					ruler = { has_variable = is_andrei_yuriyevich }
					year >= 1855
				}
				ruler = { kill_character = yes }
			}
			if = {
				limit = {
					ruler = { has_variable = is_andrei_yuriyevich }
					OR = {
						AND  = {
							year >= 1846
							scope:journal_entry = {
								is_goal_complete = yes
							}
						}
						AND = {
							year >= 1851
						}
					}
				}
				random_list = {
					11 = { }
					1 = {
						ruler = { kill_character = yes }
					}
				}
			}
		}
	}

	current_value = {
		value = var:nov_var
	}

	goal_add_value = {
		add = 3
	}

	progressbar = yes

	timeout = 7300

	weight = 10000
	should_be_pinned_by_default = yes
}

	group = je_group_internal_affairs

	complete = {
		OR = {
			NOT = { 
				has_law = law_type:law_oligarchy 
			}
			NOT = {
				any_scope_state = {
					is_incorporated = no
				}
			}
		}

		approaching_bureaucracy_shortage = no

	}

	invalid = {
		custom_tooltip = {
			text = finished_nov_tt
			has_variable = nov_complete
		}
	}

	transferable = no

	weight = 5000
}
