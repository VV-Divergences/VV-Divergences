je_ottoman_consolidation = {
	icon = "gfx/interface/icons/event_icons/waving_flag.dds"

	group = je_group_historical_content

	immediate = {
		cu:misri = {
			save_scope_as = misri_culture
		}
		cu:mashriqi = {
			save_scope_as = mashriqi_culture
		}
		c:TUR = {
			save_scope_as = ottomans_scope
		}
		set_variable = ottomans_member
	}
	on_monthly_pulse = {
		effect = {
			if = {
				limit = {
					c:TUR ?= this
					government_legitimacy >= 50
				}
                change_global_variable = {
				    name = ottoman_consolidation_var
				    add = 1
			    }
			}
		}
	}

	complete = {
		hidden_trigger = {
			c:TUR = this
		}
		scope:journal_entry = { is_goal_complete = yes }
		OR = {
			c:EGY ?= {
				is_subject_of = c:TUR
			}
			c:IRQ ?= {
				is_subject_of = c:TUR
			}
		}
		custom_tooltip = {
			text = ottomans_no_revolutions_tt
			NOT = {
				any_country = {
					is_subject_of = c:TUR
					OR = {
						country_has_primary_culture = cu:mashriqi
						country_has_primary_culture = cu:misri
					}
					has_revolution = yes
				}
			}
		}
	}

	on_complete = {
		custom_tooltip = {
			text = ottoman_consolidation_completion_tt
			if = {
				limit = {
					c:TUR ?= this
				}
				trigger_event = {
					id = ottoman_consolidation_events.1
				}
			}
		}
	}

	is_progressing = {
		c:TUR ?= {
			government_legitimacy >= 50
		}
		OR = {
			c:EGY ?= {
				is_subject_of = c:TUR
			}
			c:IRQ ?= {
				is_subject_of = c:TUR
			}
		}
		NOT = {
			any_country = {
				is_subject_of = c:TUR
				has_revolution = yes
			}
		}
	}

	fail = {
		OR = {
			NOT = {
				exists = c:TUR
			}
			AND = {
				OR = {
					NOT = {
						exists = c:EGY
					}
					c:EGY ?= {
						NOT = { is_subject_of = c:TUR }
					}
				}
				OR = {
					NOT = {
						exists = c:IRQ
					}
					c:IRQ ?= {
						NOT = { is_subject_of = c:TUR }
					}
				}
			}
			NOT = {
				country_has_primary_culture = cu:turkish
			}
			c:TUR ?= {
				is_subject = yes
			}
		}
	}

	on_fail = {
		if = { #Why? So that [somehow] this doesn't cause four fail events to fire at once. That should be impossible, but you never know.
			limit = {
				c:TUR ?= this
			} 
			if = {
				limit = {
					any_country = {
						has_variable = ottomans_member
						is_player = yes
					}
				}
				random_country = {
					limit = {
						has_variable = ottomans_member
						is_player = yes
					}
					trigger_event = { id = ottoman_consolidation_events.3 popup = yes }
				}
			}
			else = {
				ordered_country = {
					limit = {
						has_variable = ottomans_member
					}
					order_by = gdp
					trigger_event = { id = ottoman_consolidation_events.3 popup = yes }
				}
			}
		}
		else_if = {
			limit = {
				NOT = {
					exists = c:TUR
				}
			}
			if = {
				limit = {
					any_country = {
						has_variable = ottomans_member
						is_player = yes
					}
				}
				random_country = {
					limit = {
						has_variable = ottomans_member
						is_player = yes
					}
					trigger_event = { id = ottoman_consolidation_events.3 popup = yes }
				}
			}
			else = {
				ordered_country = {
					limit = {
						has_variable = ottomans_member
					}
					order_by = gdp
					trigger_event = { id = ottoman_consolidation_events.3 popup = yes }
				}
			}
		}
	}

	invalid = {
		OR = {
			AND = {
				OR = {
					c:EGY ?= this
					c:IRQ ?= this
				}
				NOT = {
					is_subject_of = c:TUR
				}
			}
			NOT = {
				exists = c:TUR
			}
			AND = {
				exists = c:TUR 
				c:TUR = {
					NOT = {
						has_journal_entry = je_ottoman_consolidation
					}
				}
			}
		}
	}

	current_value = {
		value = global_var:ottoman_consolidation_var #This variable is initialised in Argentina's country file, because of load orders.
	}

	goal_add_value = {
		value = 120
	}

	progressbar = yes

	weight = 10000

	should_be_pinned_by_default = yes
}
