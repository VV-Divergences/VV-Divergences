je_golden_liberty = {
    icon = "gfx/interface/icons/event_icons/event_broken_crown.dds"

	group = je_group_internal_affairs

    modifiers_while_active = {
        modifier_slave_to_liberty
    }

    complete = {
        OR = {
            AND = {
                ig:ig_landowners = {
			        is_powerful = yes
			        is_in_government = yes
		        }
		        has_insurrectionary_interest_groups = no
                has_law = law_type:law_monarchy
                has_law = law_type:law_agrarianism
                OR = { 
                    has_law = law_type:law_serfdom
                    has_law = law_type:law_tenant_farmers
                }
                has_law = law_type:law_oligarchy
            }
            AND = {
                ig:ig_landowners = {
			        is_powerful = no
			        is_in_government = no
		        }
                has_insurrectionary_interest_groups = no
                NOT = {
                    has_law = law_type:law_national_supremacy
                    has_law = law_type:law_ethnostate
                    has_law = law_type:law_serfdom
                    has_law = law_type:law_traditionalism
                    has_law = law_type:law_autocracy
                    has_law = law_type:law_oligarchy
                    has_law = law_type:law_landed_voting
                }
            }
        }
    }

    on_complete = {
        custom_tooltip = {
            text = dvg_on_complete_pol_magnate_tt
            if = {
                limit = {
                    ig:ig_landowners = {
                        is_powerful = yes
                    }
                }
                trigger_event = {
                    id = dvg_poland.2
                }
                add_modifier = {
                    name = modifier_magnate_oligarchy
                    months = -1
                }
            }
            else = {
                # trigger_event = {
                #     id = dvg_poland.3
                # }
                add_modifier = {
                    name = modifier_noble_democracy
                    months = -1
                }
                add_modifier = {
                    name = dvg_pol_tech_spread
                    years = 10
                }
            }
        }
    }

    timeout = 16425

    on_timeout = {
        custom_tooltip = {
            text = dvg_on_fail_pol_magnate_tt
            if = {
                limit = {
                    ig:ig_landowners = {
                        is_powerful = yes
                    }
                }
                add_modifier = {
                    name = modifier_magnate_oligarchy
                    months = -1
                }
            }
            else = {
                # trigger_event = {
                #     id = dvg_poland.3
                # }
                add_modifier = {
                    name = modifier_noble_democracy
                    months = -1
                }
            }
        }
	}

    weight = 100
	should_be_pinned_by_default = yes
}

je_uniting_the_commonwealth = {

    is_shown_when_inactive = {
        exists = c:POL

    }

        }
    }

        }
    }

            id = dvg_poland.4
        }

        }
    }

    weight = 100
	should_be_pinned_by_default = yes
}

je_eastern_resistance = {
    icon = "gfx/interface/icons/event_icons/event_military.dds"
    
	group = je_group_internal_affairs
	

    complete = { 
        NOT = { 
            has_law = law_type:law_serfdom
        }
        cu:cossack = {
            NOT = { culture_is_discriminated_in = ROOT }
        }
        cu:cossack = {
            NOT = { culture_is_discriminated_in = ROOT }
        }
        cu:cossack = {
            NOT = { culture_is_discriminated_in = ROOT }
        }
    }

    fail = {
        c:PLC = {
            is_diplomatic_play_initiator = yes
        }
        any_country = { 	
            is_secessionist = yes
            is_diplomatic_play_target = yes
            OR = {
                country_has_primary_culture = cu:cossack
                country_has_primary_culture = cu:byelorussian
                country_has_primary_culture = cu:ruthenian
            }
        }
    }

    on_fail = {
        trigger_event = {
            id = dvg_poland.5
            days = 21
        }
    }

    on_complete = {
        trigger_event = {
            id = dvg_poland.6
        }
    }

    weight = 100
	should_be_pinned_by_default = yes
}
